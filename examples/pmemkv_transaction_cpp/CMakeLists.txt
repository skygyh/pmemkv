cmake_minimum_required(VERSION 3.3)
project(pmemkv_transaction_cpp CXX)

find_package(PkgConfig REQUIRED)
pkg_check_modules(LIBPMEMKV REQUIRED libpmemkv)

include_directories(${LIBPMEMKV_INCLUDE_DIRS})
link_directories(${LIBPMEMKV_LIBRARY_DIRS})
add_executable(pmemkv_transaction_cpp pmemkv_transaction.cpp)
target_link_libraries(pmemkv_transaction_cpp ${LIBPMEMKV_LIBRARIES})
